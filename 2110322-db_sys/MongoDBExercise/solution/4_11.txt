db.customer.aggregate(
  [
    {
      $lookup: {
        from: 'order',
        localField: 'customer_id',
        foreignField: 'customer_id',
        as: 'info'
      }
    },
    { $match: { info: { $size: 0 } } },
    {
      $project: {
        _id: 0,
        customer_id: 1,
        customer_name: 1
      }
    },
    { $sort: { customer_id: 1 } }
  ]
)
